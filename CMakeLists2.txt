cmake_minimum_required(VERSION 3.10)
project(MyProject)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add the server and client subdirectories
#add_subdirectory(server)
add_subdirectory(client)

# Add other library subdirectories
add_subdirectory(lib)
add_subdirectory(lib_Test)
add_subdirectory(lib_logger)
add_subdirectory(nlohmann)
add_subdirectory(sqlite3)

# Link libraries to a shared target
add_library(shared_libraries INTERFACE)
foreach(LIBRARY ${LIBRARIES})
    target_link_libraries(shared_libraries INTERFACE ${LIBRARY})
endforeach()

# Find and link OpenCV
find_package(OpenCV REQUIRED)
target_link_libraries(shared_libraries INTERFACE ${OpenCV_LIBS})

set(ASSETS_DIR ${CMAKE_CURRENT_SOURCE_DIR}/assets)
